1    // A Fascinating number is one which when multiplied by 2 and 3 and then, after the results are concatenated
2    // with the original number, the new number contains all the digits from 1 to 9 exactly once. There can be any number of zeroes and are to be ignored.

7    import java.util.Arrays;
8    import java.util.Scanner;

10   public class fascinatingNumber {
11       public static void main(String[] args) {
12           Scanner sc = new Scanner(System.in);
13           System.out.println("Enter lower range: ");
14           int l = sc.nextInt();
15           System.out.println("Enter higher range: ");
16           int h = sc.nextInt();
17           sc.close();
18           int count = 0;
19           for (int i = l; i <= h; i++) {
20               if (isFascinating(i)) {
21                   System.out.print(i + " ");
22                   count += 1;
23               }
24           }
25           System.out.println("\nNo of fascinating numbers found: " + count);
26       }

28       static boolean isFascinating(int no) {
29           int two = no * 2;
30           int three = no * 3;
31           String str = "" + no + two + three;
32           // Removing all zeroes from number
33           String tmp = str;
34           str = "";
35           for (int i = 0; i < tmp.length(); i++) {
36               char ch = tmp.charAt(i);
37               if (ch != '0')
38                   str += ch;
39           }
40           // Sorting digits in ascending order
41           char[] digits = str.toCharArray();
42           Arrays.sort(digits);
43           String sortedDigits = String.valueOf(digits);
44           return sortedDigits.equals("123456789");
45       }
